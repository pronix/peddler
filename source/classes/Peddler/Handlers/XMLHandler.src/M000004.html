      def self.parse_legacy(xml)
        if xml["Batch"]
          xml["Batch"].collect  { |input| Peddler::LegacyReports::UploadStatus.new(input) }
        elsif xml["Report"]
          xml["Report"].collect { |input| Peddler::LegacyReports::ReportStatus.new(input) }
        end
      end
