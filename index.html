peddler


Peddler is a Ruby wrapper to the Amazon Inventory Management API. 
$ irb -r "lib/peddler"
>> @client = Peddler::Client.new
>> @client.username = "john@example.com"
>> @client.password = "secret"
>> @client.region = "us"
>>
>> @client.inventory_queue_count
=> 0
>>
>> batch = @client.new_inventory_batch
>> batch.id
=> nil
>>
>> item = Peddler::Inventory::Item.new
>> item.product_id = "1234567890"
>> item.price = 100.00
>> item.sku = "FOO-SKU"
>> item.quantity = 10
>>
>> batch << item
>> batch.upload
=> true
>> batch.id
=> "2587134052"
>>
>> latest_batches = @client.latest_inventory_uploads :count => 1
=> [#<Peddler::Inventory::BatchStatus batch_id="2587104060", status="In Progress">]
>> last_batch = last_batches[0]
>> last_batch.status
=> "In Progress"
>>
>> report = @client.new_report :upload, 2587104060
>> report.response
=> "Feed Processing Summary:\n\tNumber of records processed\t\t1\n\tNum...
>>
