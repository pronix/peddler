peddler


Peddler is a Ruby wrapper to the Amazon Inventory Management API. 

$ irb -r "lib/peddler"
>> client = Peddler::Client.new(
?>   :username => "foo@bar.com",
?>   :password => "secret",
?>   :region   => "us")
>> batch = client.new_inventory_batch
>> item = client.new_inventory_item(
?>   :product_id => "1234567890",
?>   :price      => 100.00,
?>   :sku        => "SKU-123",
?>   :quantity   => 10)
>> batch << item
>> batch.upload
>> batch.id
=> "9876543210"
>> client.inventory_queue_count
=> 1
>> sleep(60)
>> log = client.latest_inventory_uploads(
?>   :count => 1)
>> log.items_activated
=> 1
>> log.batch_id
=> "9876543210"
