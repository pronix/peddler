    def query_string
      unless query_params.empty?
        query_params.inject("?") do |out, pair|
          key, value = pair
          key = key.to_s.gsub(/_([a-z])/) { $1.upcase } if key.kind_of? Symbol
          value = value.httpdate if value.respond_to? :httpdate
          out += "&" if out.size > 1
          "#{out}#{url_encode(key)}=#{url_encode(value)}"
        end
      end
    end
