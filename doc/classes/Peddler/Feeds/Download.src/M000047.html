      def initialize(transport, params={})
        @mapped_params = {
          "DownloadId"          => "id",
          "DownloadType"        => "type",
          "RelatedReferenceId"  => "related_reference_id",
          "AvailableDate"       => "available_at",
          "Acknowledged"        => "acknowledged"}
        @transport = transport
        params.each_pair{ |key, value| self.send "#{@mapped_params[key]}=", value }
      end
