      def self.parse(name, xml)
        name = name.to_s.capitalize
        list = xml["#{name}sStatusList"] || xml["#{name}sList"]
        if list
          list.collect { |s| parse_status(name, s) }
        else
          [ parse_status(name, xml) ]
        end
      end
